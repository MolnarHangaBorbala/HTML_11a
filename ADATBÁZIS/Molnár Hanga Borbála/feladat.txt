SELECT meresek.ev, SUM(meresek.perc) / 60 AS orak
FROM meresek INNER JOIN regiok on meresek.regioId=regiok.ID
WHERE regiokNev='Anglia'
GROUP BY meresek.ev
HAVING meresek.ev BETWEEN 1990 AND 2000
ORDER BY meresek.ev DESC
